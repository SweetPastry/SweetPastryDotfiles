\spChapter[复变函数]{复变函数}
    \section{Cauchy 积分公式}
        \begin{theorem}[Cauchy 积分公式]
            设 $f\left(z\right)$ 在区域 $D$ 解析, 且 $\partial D$ 连续, 则对于任意的 $z_0\in D$ 都有
            $$
            f\left(z_0\right) = \oint_{\partial D}{\dfrac{f\left(z\right)}{z-z_0}.\mathrm{d}z}
            $$
        \end{theorem}
